ESP.lua

Wallhack script that shows player locations.

-- ESP Script
for i,v in pairs(game.Players:GetPlayers()) do
  if v ~= game.Players.LocalPlayer then
    local esp = Instance.new("BoxHandleAdornment", v.Character)
    esp.Adornee = v.Character:WaitForChild("HumanoidRootPart")
    esp.AlwaysOnTop = true
    esp.ZIndex = 10
    esp.Size = Vector3.new(4, 6, 1)
    esp.Color3 = Color3.new(1, 0, 0)
  end
end
